## Solution

```c++
#include <bits/stdc++.h>
// #define int long long
#define double long double
#define V vector
#define NL '\n'
#define sz(x) ((int)x.size())
using namespace std;

int uf[2001 * 2001];
int szs[2001 * 2001];

int find(int i) {
  while (i != uf[i]) {
    uf[i] = uf[uf[i]];
    i = uf[i];
  }
  return i;
}

bool connect(int x, int y) {
  x = find(x);
  y = find(y);
  if (x == y) {
    return false;
  }
  if (szs[x] < szs[y]) {
    szs[y] += szs[x];
    uf[x] = y;
  } else {
    szs[x] += szs[y];
    uf[y] = x;
  }
  return true;
}

signed main(void) {
  cin.tie(0)->sync_with_stdio(0);
  freopen("fencedin.in", "r", stdin);
  freopen("fencedin.out", "w", stdout);
  // Flipped because it's more natural for me.
  int a, b, m, n;
  cin >> a >> b >> m >> n;
  V<int> vertical = {0, a};
  for (int i = 0; i < m; i++) {
    int x;
    cin >> x;
    vertical.push_back(x);
  }
  V<int> horizontal = {0, b};
  for (int i = 0; i < n; i++) {
    int x;
    cin >> x;
    horizontal.push_back(x);
  }
  sort(vertical.begin(), vertical.end());
  sort(horizontal.begin(), horizontal.end());
  V<array<int, 3>> edges;
  for (int i = 0; i < n; i++) {
    for (int j = 0; j <= m; j++) {
      edges.push_back({i * (m + 1) + j, (i + 1) * (m + 1) + j,
                       vertical[j + 1] - vertical[j]});
    }
  }
  for (int i = 0; i < m; i++) {
    for (int j = 0; j <= n; j++) {
      edges.push_back({j * (m + 1) + i, j * (m + 1) + i + 1,
                       horizontal[j + 1] - horizontal[j]});
    }
  }
  auto comp = [](array<int, 3> &x, array<int, 3> &y) { return x[2] < y[2]; };
  sort(edges.begin(), edges.end(), comp);
  int nodes = (n + 1) * (m + 1);
  for (int i = 0; i < nodes; i++) {
    uf[i] = i;
    szs[i] = 1;
  }
  int mst_edges = 0;
  int edge_index = 0;
  long long sum = 0;
  while (mst_edges < nodes - 1) {
    auto edge = edges[edge_index++];
    if (connect(edge[0], edge[1])) {
      sum += edge[2];
      mst_edges++;
    }
  }
  cout << sum << NL;
}
```
