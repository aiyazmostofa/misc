## Solution

```c++
#include <bits/stdc++.h>
#define int long long
#define double long double
#define V vector
#define NL '\n'
#define sz(x) ((int)x.size())
using namespace std;

int v[100];
int n, k;

int dp[101][101][7];
const int UNKNOWN = -1;
const int ERROR = -2;

int rec(int i, int offset, int open) {
  if (dp[i][offset][open] != UNKNOWN) {
    return dp[i][offset][open];
  }
  if (i == n) {
    return dp[i][offset][open] = 0;
  }
  if (open == 0) {
    return dp[i][offset][open] = ERROR;
  }
  int mn = ERROR;
  int sum = 0;
  for (int j = i; j < n; j++) {
    sum += v[(j + offset) % n] * (j - i);
    int result = rec(j + 1, offset, open - 1);
    if (result == ERROR) {
      continue;
    }
    result += sum;
    if (mn == ERROR || result < mn) {
      mn = result;
    }
  }
  return dp[i][offset][open] = mn;
}

signed main(void) {
  cin.tie(0)->sync_with_stdio(0);
  freopen("cbarn2.in", "r", stdin);
  freopen("cbarn2.out", "w", stdout);
  cin >> n >> k;
  for (int i = 0; i < n; i++) {
    cin >> v[i];
  }
  for (int i = 0; i < 101 * 101 * 7; i++) {
    ((int *)dp)[i] = UNKNOWN;
  }
  int mn = ERROR;
  for (int i = 0; i < n; i++) {
    int result = rec(0, i, k);
    if (result == ERROR) {
      continue;
    }
    if (mn == ERROR || result < mn) {
      mn = result;
    }
  }
  cout << mn << NL;
}
```
